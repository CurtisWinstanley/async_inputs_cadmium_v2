155  void addCoupling(const std::shared_ptr<PortInterface>& portFrom, const std::shared_ptr<PortInterface>& portTo) {
156  if (!portTo->compatible(portFrom)) {
157  throw CadmiumModelException("invalid port type");
158  }
159  if (portFrom->getParent() == nullptr || portTo->getParent() == nullptr) {
160  throw CadmiumModelException("port does not belong to any model");
161  }
162  auto portFromParent = portFrom->getParent();
163  auto portToParent = portTo->getParent();
164  if (inPorts.containsPort(portFrom)) {
165  if (portToParent->getParent() == this && portToParent->containsInPort(portTo)) {
166  addCoupling(EIC, portFrom, portTo);
167  } else {
168  throw CadmiumModelException("invalid destination port");
169  }
170  } else if (portFromParent->getParent() == this && portFromParent->containsOutPort(portFrom)) {
171  if (outPorts.containsPort(portTo)) {
172  addCoupling(EOC, portFrom, portTo);
173  } else if (portToParent->getParent() == this && portToParent->containsInPort(portTo)) {
174  addCoupling(IC, portFrom, portTo);
175  } else {
176  throw CadmiumModelException("invalid destination port");
177  }
178  } else {
179  throw CadmiumModelException("invalid origin port");
180  }
181  }
