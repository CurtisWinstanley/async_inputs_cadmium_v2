118  void addCouplings() {
119  for (const auto& cell: Coupled::components) {
120  auto cellModel = std::dynamic_pointer_cast<Cell<C, S, V>>(cell);
121  if (cellModel == nullptr) {
122  throw CadmiumModelException("Scenario component is not a cell");
123  }
124  for (const auto& neighbor: cellModel->getNeighborhood()) {
125  addIC(cellId(neighbor.first), "neighborhoodOutput",
126  cellModel->getId(), "neighborhoodInput");
127  }
128  auto cellConfig = cellModel->getCellConfig();
129  for (const auto& [portFrom, portTo]: cellConfig->EIC) {
130  addEIC(portFrom, cellModel->getId(), portTo);
131  }
132  for (const auto& portTo: cellConfig->EOC) {
133  addEOC(cellModel->getId(), "neighborhoodOutput", portTo);
134  }
135  }
136  }
